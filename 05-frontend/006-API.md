# Frontend - API Integration

## API Client

O cliente Axios centraliza todas as requisições HTTP.

### Configuração Base

```typescript
// api/client.ts
import axios from 'axios';

const baseURL = import.meta.env.VITE_API_BASE_URL || 'http://localhost:3000/api/v1';

export const apiClient = axios.create({
  baseURL,
  timeout: 5000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Add JWT token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('auth_token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Handle errors
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('auth_token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);
```

## API Functions

### Bans API

```typescript
// api/bans.api.ts
import apiClient from './client';

export const bansApi = {
  listByGuild: async (guildId: string, page = 1, limit = 50) => {
    const response = await apiClient.get('/moderation/bans', {
      params: { guildId, page, limit },
    });
    return response.data;
  },

  create: async (data: {
    guildId: string;
    userId: string;
    reason: string;
    moderatorId: string;
    expiresAt?: string;
  }) => {
    const response = await apiClient.post('/moderation/bans', data);
    return response.data.data;
  },

  remove: async (banId: string) => {
    const response = await apiClient.delete(`/moderation/bans/${banId}`);
    return response.data;
  },
};
```

### Guilds API

```typescript
// api/guilds.api.ts
import apiClient from './client';

export const guildsApi = {
  list: async () => {
    const response = await apiClient.get('/core/guilds');
    return response.data;
  },

  getById: async (guildId: string) => {
    const response = await apiClient.get(`/core/guilds/${guildId}`);
    return response.data;
  },

  getStats: async (guildId: string) => {
    const response = await apiClient.get(`/core/guilds/${guildId}/stats`);
    return response.data;
  },

  updateConfig: async (guildId: string, config: object) => {
    const response = await apiClient.patch(`/config/guilds/${guildId}`, config);
    return response.data;
  },
};
```

## Custom Hooks

### useApi Hook

```typescript
// hooks/useApi.ts
import { useState, useCallback } from 'react';

interface UseApiState<T> {
  data: T | null;
  isLoading: boolean;
  error: string | null;
}

export function useApi<T>() {
  const [state, setState] = useState<UseApiState<T>>({
    data: null,
    isLoading: false,
    error: null,
  });

  const execute = useCallback(async (apiCall: () => Promise<T>) => {
    setState({ data: null, isLoading: true, error: null });
    try {
      const data = await apiCall();
      setState({ data, isLoading: false, error: null });
      return data;
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Unknown error';
      setState({ data: null, isLoading: false, error: message });
      throw error;
    }
  }, []);

  return { ...state, execute };
}
```

## Error Handling

### Tratamento Centralizado

```typescript
// utils/api.utils.ts
export function handleApiError(error: unknown): string {
  if (axios.isAxiosError(error)) {
    if (error.response) {
      // Server responded with error
      return error.response.data?.message || `Error ${error.response.status}`;
    }
    if (error.request) {
      // No response received
      return 'Network error. Please try again.';
    }
  }
  return 'An unexpected error occurred';
}
```

### Uso em Componentes

```typescript
try {
  await bansApi.create(data);
} catch (error) {
  const message = handleApiError(error);
  toast.error(message);
}
```
